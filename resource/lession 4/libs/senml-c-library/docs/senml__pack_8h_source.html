454  inline char readHexChar(Stream *source){
455  #ifdef __MBED__
456  unsigned char first = source->getc();
457  unsigned char second = source->getc();
458  #else
459  unsigned char first = source->read();
460  unsigned char second = source->read();
461  #endif
462  first = (first < '9') ? first - '0' : first - '7';
463  second = (second < '9') ? second - '0' : second - '7';
464  return (16 * first) + second;
